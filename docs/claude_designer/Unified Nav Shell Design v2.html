01
v1 → v2 · why 3 modes
Read first

v1 designed a sidebar. It's correct as far as it went — 10 items, grouped EDIT / ASSETS / TOOLS, clean iframes for the wrappers and a native panel for Forge. But it answered the wrong question. Your system has three jobs that share almost no UI:

ModeJobVerbOutput
STUDIOCut the frag movieedit.avi file per part
LABWork the demo pileextract / pattern-find / annotateclip candidates in DB
CREATIVE SUITEGenerate the artprompt / render / packpk3 style packs

Each feeds the next: LAB produces clips → STUDIO cuts them → CREATIVE supplies the skins/textures STUDIO renders them with. Cramming all three into one flat sidebar buries the model. v2 promotes the 3 jobs to a top-level mode switch. Same URL, same store, same chrome — but each mode has its own sidebar. Matches how you described the dream: "the Lab where we work on demo extraction ... the Creative Suite where we test skins, sprites, prompt to generate our maps."

Nothing new gets built at the API layer. All 14 routers you shipped are already sufficient; v2 is pure frontend wiring — a mode switch in the header, a nested PAGE_MAP in studio-pages.js, 15 new panel modules (mostly thin iframes or list views over endpoints that exist). Claude Code session is ~4 hours end-to-end.

StudioStore extensions
+1 field · +1 action · +1 action for part swap preservation
// studio-store.js — INITIAL_STATE
var INITIAL_STATE = {
  activeMode:    'studio',        // 'studio' | 'lab' | 'creative'
  modePage:      {                // last page per mode — preserved across switches
    studio:   'preview',
    lab:      'demos',
    creative: 'textures',
  },
  activePage:    'preview',
  activePart:    null,
  parts:         [],
  clips:         [],
  isPlaying:     false,
  currentTime:   0,
  buildStatus:   null,
  statusMessage: 'Ready',
};

// New action — dispatched by the mode-switch buttons
case 'SET_ACTIVE_MODE': {
  var nextMode = action.payload;
  var nextPage = _state.modePage[nextMode] || _DEFAULTS[nextMode];
  setState({
    activeMode: nextMode,
    activePage: nextPage,
  });
  break;
}

// Update SET_ACTIVE_PAGE to remember the page per mode
case 'SET_ACTIVE_PAGE': {
  var page = action.payload;
  var mp = Object.assign({}, _state.modePage);
  mp[_state.activeMode] = page;
  setState({ activePage: page, modePage: mp });
  setState({ activePage: action.payload });
  break;
}

// Defaults constant (module-private)
var _DEFAULTS = { studio: 'preview', lab: 'demos', creative: 'textures' };

Why preserve page per mode: flip from STUDIO-Timeline → LAB-Patterns → back to STUDIO should land you on Timeline, not Preview. This is 6 lines and saves the user's muscle memory.

  // ── Panel swap ─────────────────────────────────────────────────────────
  function _switch(mode, page) {
    var key = mode + ':' + page;
    if (key === _currentKey) return;
    var slot = document.getElementById('panel-slot');
    if (!slot) return;

    if (_currentPanel && typeof _currentPanel.unmount === 'function') {
      try { _currentPanel.unmount(); } catch (e) { console.error('[Pages] unmount', e); }
    }
    _currentPanel = null;
    slot.replaceChildren();

    var cfg = _lookup(mode, page);
    if (!cfg) { _placeholder(slot, page); _currentKey = key; return; }
    var mod = global[cfg.module];
    if (!mod || typeof mod.mount !== 'function') { _placeholder(slot, page); _currentKey = key; return; }

    try { mod.mount(slot); _currentPanel = mod; }
    catch (e) { console.error('[Pages] mount', e); _placeholder(slot, page); }

    _currentKey = key;
    _syncStatusChip(mode);
  }

  function _lookup(mode, page) {
    var groups = (NAV[mode] || {}).groups || [];
    for (var i = 0; i < groups.length; i++)
      for (var j = 0; j < groups[i].items.length; j++)
        if (groups[i].items[j].page === page) return groups[i].items[j];
    return null;
  }

  function _placeholder(slot, page) {
    var d = document.createElement('div');
    d.className = 'panel-not-loaded';
    d.textContent = 'Panel not loaded: ' + page;
    slot.replaceChildren(d);
  }

  function _syncStatusChip(mode) {
    var el = document.getElementById('status-mode-chip');
    if (!el) return;
    el.className = 'status-mode ' + mode;
    el.textContent = mode.toUpperCase();
  }

  // ── Querystring sync ───────────────────────────────────────────────────
  function _syncUrl(mode, page) {
    var u = new URL(window.location.href);
    u.searchParams.set('mode', mode);
    u.searchParams.set('page', page);
    window.history.replaceState(null, '', u.toString());
  }

  function _readUrl() {
    var u = new URL(window.location.href);
    return { mode: u.searchParams.get('mode'), page: u.searchParams.get('page') };
  }

  // ── Event wiring ───────────────────────────────────────────────────────
  function _wire() {
    // mode switch
    document.querySelectorAll('#mode-switch .mode-btn').forEach(function (b) {
      b.addEventListener('click', function () {
        var m = b.getAttribute('data-mode');
        global.StudioStore.dispatch({ type: 'SET_ACTIVE_MODE', payload: m });
        document.querySelectorAll('#mode-switch .mode-btn').forEach(function (x) {
          x.classList.toggle('active', x === b);
          x.setAttribute('aria-selected', x === b ? 'true' : 'false');
        });
      });
    });

    // sidebar click/keyboard
    var sidebar = document.getElementById('sidebar');
    sidebar.addEventListener('click', function (e) {
      var item = e.target.closest('.nav-item'); if (!item) return;
      var page = item.getAttribute('data-page');
      global.StudioStore.dispatch({ type: 'SET_ACTIVE_PAGE', payload: page });
    });
    sidebar.addEventListener('keydown', function (e) {
      if (e.key !== 'Enter' && e.key !== ' ') return;
      var item = e.target.closest('.nav-item'); if (!item) return;
      e.preventDefault();
      var page = item.getAttribute('data-page');
      global.StudioStore.dispatch({ type: 'SET_ACTIVE_PAGE', payload: page });
    });
  }

  function init() {
    var store = global.StudioStore;
    if (!store) { console.error('[Pages] no StudioStore'); return; }

    // 1. Consume URL if present
    var q = _readUrl();
    if (q.mode)  store.dispatch({ type: 'SET_ACTIVE_MODE', payload: q.mode  });
    if (q.page)  store.dispatch({ type: 'SET_ACTIVE_PAGE', payload: q.page  });

    _wire();

    // 2. Subscribe
    store.subscribe(function (s, p) {
      if (s.activeMode !== p.activeMode) {
        _renderSidebar(s.activeMode, s.activePage);
        document.querySelectorAll('#mode-switch .mode-btn').forEach(function (b) {
          b.classList.toggle('active', b.getAttribute('data-mode') === s.activeMode);
        });
      }
      if (s.activePage !== p.activePage || s.activeMode !== p.activeMode) {
        _syncActive(s.activePage);
        _switch(s.activeMode, s.activePage);
        _syncUrl(s.activeMode, s.activePage);
      }
    });

    // 3. Initial render
    var st = store.getState();
    _renderSidebar(st.activeMode, st.activePage);
    _switch(st.activeMode, st.activePage);
    _syncUrl(st.activeMode, st.activePage);
  }

  function _syncActive(page) {
    document.querySelectorAll('#nav-list .nav-item').forEach(function (it) {
      it.classList.toggle('active', it.getAttribute('data-page') === page);
    });
  }

  global.StudioPages = { init: init };

  if (document.readyState === 'loading')
    document.addEventListener('DOMContentLoaded', init);
  else
    init();

}(window));
